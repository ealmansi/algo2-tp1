\begin{tad}{\tadNombre{Gremio}}

\tadIgualdadObservacional{g}{g'}{gremio}
{nombre(g) $\igobs$ nombre(g')\mand \\
trabajadores(g) $\igobs$ trabajadores(g')\mand \\
empresas(g) $\igobs$ empresas(g') }	

\tadGeneros{gremio}

\tadExporta{gremio, generadores, observadores, cantTrabajadores, obtenerNombres}

\tadUsa{\tadNombre{Bool}, \tadNombre{Nat}, \tadNombre{Conjunto($\alpha$)}}
	
\tadGeneradores
	\tadOperacion{CrearGremio}{nombre/n,conj(trabajador)/ct,conj(empresa)/ce}{gremio}{\mnot \esvacio{ct}\mand \mnot \esvacio{ce}}
	
\tadObservadores
	\tadAlinearFunciones{trabajadores}{gremio}
	\tadOperacion{nombre}{gremio}{nombre}{}
	\tadOperacion{trabajadores}{gremio}{conj(trabajador)}{}
	\tadOperacion{empresas}{gremio}{conj(empresa)}{}		

\tadOtrasOperaciones
	\tadAlinearFunciones{cantTrabajadores}{gremio}
	\tadOperacion{cantTrabajadores}{gremio}{nat}{}

	\vspace{2mm}
	\tadAlinearFunciones{obtenerNombres}{conj(gremio)}
 	\tadOperacion{obtenerNombres}{conj(gremio)}{conj(nombre)}{}

\tadAxiomas{\paratodo{n}{nat}, \paratodo{ct}{conj(trabajador)}, \paratodo{ce}{conj(empresa)}, \paratodo{g}{gremio}} \\

	\tadAlinearAxiomas{trabajadores(CrearGremio(n, ct, ce))}
	\tadAxioma{nombre(CrearGremio(n, ct, ce))}{n}
	\tadAxioma{trabajadores(CrearGremio(n, ct, ce))}{ct}
	\tadAxioma{empresas(CrearGremio(n, ct, ce))}{ce}

	\vspace{2mm}
	\tadAlinearAxiomas{cantTrabajadores(g)}
	\tadAxioma{cantTrabajadores(g)}{\#(trabajadores(g))}

	\vspace{2mm}
 	\tadAlinearAxiomas{obtenerNombres(cg)}
 	\tadAxioma{obtenerNombres(cg)}{
		\iffi{\esvacio{cg}}
		{\vacio}
		{Ag(nombre(dameUno(cg)), obtenerNombres(sinUno(cg)))}
	}
		
\end{tad}
