\begin{tad}{\tadNombre{ConfederacionGremios}}

\tadIgualdadObservacional{c}{c'}{confederacion}{gremios(c) \migobs gremios(c') \mandl \\
(\paratodoC{ng}{nombre}{ng \en obtenerNombres(gremios(c))}) \\ aliados(ng,c) \migobs aliados(ng,c')
}	

\tadGeneros{confederacion}

\tadExporta{confederacion, observadores}

\tadUsa{\tadNombre{Nombre}, \tadNombre{Gremio}, \tadNombre{Conjunto($\alpha$)}}

\tadObservadores
	\tadAlinearFunciones{aliados}{nombre/ng,confederacion/c}
 	\tadOperacion{gremios}{confederacion/c}{conj(gremio)}{}
 	\tadOperacion{aliados}{nombre/ng,confederacion/c}{conj(nombre)}{ng \en obtenerNombres(gremios(c))}

\tadGeneradores
	\tadAlinearFunciones{CrearConfederacion}{}
	\tadOperacion{CrearConfederacion}{}{confederacion}{}
	\tadAlinearFunciones{CrearConfederacion}{nombre/ng1,nombre/ng2,confederacion/c}
	\tadOperacion{AgregarGremio}{gremio/g,confederacion/c}{confederacion}{}
	\begin{itemize}
		\itemindent80mm
		\item[] \{ ((\paratodoC{og}{gremio}{og \en gremios(c)})
		\itemindent83mm
		\item[] (nombre(g) \mneq nombre(og) \mand
		\item[] \esvacio{empresas(g) \intersec empresas(og)} \mand
		\item[] \esvacio{trabajadores(g) \intersec trabajadores(og)})) \}
	\end{itemize}
	\tadOperacion{EstablecerAlianza}{nombre/ng1,nombre/ng2,confederacion/c}{confederacion}{}
	\begin{itemize}
		\itemindent80mm
		\item[] \{ ng1 \en obtenerNombres(gremios(c)) \mand
		\itemindent83mm
		\item[] ng2 \en obtenerNombres(gremios(c)) \mand
		\item[] ng1 \mnotin aliados(ng2,c)\}
	\end{itemize}

\tadAxiomas{\paratodo{c}{confederacion}, \paratodo{g}{gremio}, \paratodo{ng1}{nombre}, \paratodo{ng2}{nombre}} \\
	\tadAlinearAxiomas{gremios(EstablecerAlianza(ng1,ng2,c))}
  	\tadAxioma{gremios(CrearConfederacion)}{\vacio}
  	\tadAxioma{gremios(AgregarGremio(g,c))}{Ag(g, gremios(c)) }
  	\tadAxioma{gremios(EstablecerAlianza(ng1,ng2,c))}{gremios(c)}

	\vspace{2mm}
 	\tadAlinearAxiomas{aliados(ng, EstablecerAlianza(ng1,ng2,c))}
 	\tadAxioma{aliados(ng, CrearConfederacion)}{\vacio}
  	\tadAxioma{aliados(ng, AgregarGremio(g,c))}{aliados(ng,c)}
  	\tadAxioma{aliados(ng, EstablecerAlianza(ng1,ng2,c))}{
	  	\iffi{ng \meq ng1}
	  	{Ag(ng2, aliados(ng2,c))\union aliados(ng,c)}
	  	{\iffiI{ng \meq ng2}{Ag(ng1, aliados(ng1,c))\union aliados(ng,c)}{aliados(ng,c)}}
  	}

\end{tad}
